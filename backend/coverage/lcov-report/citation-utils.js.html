// Citation extraction and linkification for AI responses.
// JS port of vue-project/src/utils/citation.ts so the logic can be unit-tested
// with node --test under c8. Keep behavior identical to the TS twin.
 
const CN_DIGIT = {
  零: 0, '〇': 0, 一: 1, 二: 2, 两: 2, 三: 3, 四: 4, 五: 5,
  六: 6, 七: 7, 八: 8, 九: 9, 十: 10, 百: 100, 千: 1000
};
 
function cnToNumber(raw) {
  if (!raw) return null;
  const s = String(raw).trim();
  if (/^\d+$/.test(s)) {
    const n = parseInt(s, 10);
    return Number.isFinite(n) ? n : null;
  }
  if (!/^[零〇一二三四五六七八九十百千两]+$/.test(s)) return null;
 
  if (s === '十') return 10;
  if (s.startsWith('十') && s.length === 2) {
    const u = CN_DIGIT[s[1]];
    return u != null && u < 10 ? 10 + u : null;
  }
 
  let total = 0;
  let current = 0;
  for (const ch of s) {
    const v = CN_DIGIT[ch];
    if (v == null) return null;
    if (v === 10 || v === 100 || v === 1000) {
      current = current === 0 ? 1 : current;
      total += current * v;
      current = 0;
    } else {
      current = v;
    }
  }
  total += current;
  return total > 0 && total <= 999 ? total : null;
}
 
const CITATION_RE = /第([零〇一二三四五六七八九十百千两\d]+)章/g;
 
function extractCitations(text) {
  if (!text) return [];
  const out = [];
  let m;
  CITATION_RE.lastIndex = 0;
  while ((m = CITATION_RE.exec(text)) !== null) {
    const num = cnToNumber(m[1]);
    if (num == null || num < 1 || num > 81) continue;
    out.push({
      start: m.index,
      end: m.index + m[0].length,
      raw: m[0],
      chapter: num
    });
  }
  return out;
}
 
const CODE_SPAN_RE = /(```[\s\S]*?```|`[^`\n]*`)/g;
 
function transformPlain(chunk, basePath) {
  if (!chunk) return '';
  const citations = extractCitations(chunk);
  if (citations.length === 0) return chunk;
  let out = '';
  let cursor = 0;
  for (const c of citations) {
    out += chunk.slice(cursor, c.start);
    out += `[${c.raw}](${basePath}?chapter=${c.chapter})`;
    cursor = c.end;
  }
  out += chunk.slice(cursor);
  return out;
}
 
function linkifyCitations(text, basePath = '/knowledge-graph') {
  if (!text) return '';
  const segments = [];
  let lastIndex = 0;
  let codeMatch;
  CODE_SPAN_RE.lastIndex = 0;
  while ((codeMatch = CODE_SPAN_RE.exec(text)) !== null) {
    segments.push(transformPlain(text.slice(lastIndex, codeMatch.index), basePath));
    segments.push(codeMatch[0]);
    lastIndex = codeMatch.index + codeMatch[0].length;
  }
  segments.push(transformPlain(text.slice(lastIndex), basePath));
  return segments.join('');
}
 
module.exports = {
  cnToNumber,
  extractCitations,
  linkifyCitations
};
